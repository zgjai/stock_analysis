JavaScript代码片段（可在控制台执行）:
// 强制隐藏全局加载状态
UXUtils.hideGlobalLoading();

// 清理所有可能的遮罩层
document.querySelectorAll('#global-loading-overlay').forEach(el => el.remove());
document.querySelectorAll('.modal-backdrop').forEach(el => el.remove());
document.body.classList.remove('modal-open');

// 检查是否还有加载状态
console.log('Loading overlay exists:', !!document.getElementById('global-loading-overlay'));

// 如果交易记录管理器存在，强制隐藏加载状态
if (window.tradingManager && tradingManager.forceHideGlobalLoading) {
    tradingManager.forceHideGlobalLoading();
}
